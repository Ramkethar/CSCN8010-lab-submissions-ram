import matplotlib.pyplot as mpb

# Sample data
gender_labels = ['Male', 'Female']
gender_counts = [45, 55]  # These numbers represent the counts or percentages of males and females.

# Create a pie chart
mpb.figure(figsize=(6, 6))
mpb.pie(gender_counts, labels=gender_labels, autopct='%1.1f%%', startangle=140)
mpb.title('Male vs. Female Ratio')

# Display the pie chart
mpb.axis('equal')  # Equal aspect ratio ensures that pie is drawn as a circle.

mpb.show()

gender_labels = the labels for the two categories, "Male" and "Female."

gender_counts = the corresponding counts or percentages.

mpb.xlabel('Date')
mpb.ylabel('Daily New Cases')
mpb.title('Daily New COVID-19 Cases Trend (Example)')

import matplotlib.pyplot as mpb

# Sample data
countries = ['Canada', 'India']
population = [37590000, 1393409038]  # Population figures are in millions

# Create a bar chart
mpb.figure(figsize=(8, 6))
mpb.bar(countries, population, color=['blue', 'orange'])
mpb.xlabel('Country')
mpb.ylabel('Population (Millions)')
mpb.title('Population Comparison: Canada vs. India')

mpb.show()

countries = the names of the two countries, Canada and India. population = the corresponding population sizes in millions.
